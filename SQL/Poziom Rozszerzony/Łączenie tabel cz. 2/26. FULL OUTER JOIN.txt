SELECT title, last_name, first_name, name FROM authors LEFT JOIN books on authors.id = author_id LEFT JOIN publishers ON publishers.id = publisher_id UNION SELECT title, last_name, first_name, name FROM books LEFT JOIN authors on authors.id = author_id LEFT JOIN publishers ON publishers.id = publisher_id UNION SELECT title, last_name, first_name, name FROM publishers LEFT JOIN books ON publishers.id = publisher_id LEFT JOIN authors on authors.id = author_id  ORDER BY last_name, first_name, name, title;
